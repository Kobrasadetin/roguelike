6   package com.majesticbit.roguelike.gui;
7   
8   import com.majesticbit.roguelike.domain.Game;
9   import com.majesticbit.roguelike.domain.GameInterface;
10  import com.majesticbit.roguelike.domain.creatures.actions.Action;
11  import com.majesticbit.roguelike.domain.level.Level;
12  
13  /**
14   *
15   * @author Master
16   */
17  public class UserInterface implements GameInterface {
18  
19      private VisualInterface visualInterface;
20      private PlayerInput input;
21      private Game game;
22  
23      public UserInterface(Game game) {
24          addSwingViewer();
25          this.game = game;
26          game.addObserver(this);
27      }
28  
29      private void addSwingViewer() {
30          SwingViewer viewer = new SwingViewer();
31          SwingPlayerInput input = new SwingPlayerInput(viewer);
32          this.visualInterface = viewer;
33          this.input = input;
34          viewer.addKeyEventListener(input);
35      }
36  
37      private void draw(Level level) {
38          visualInterface.draw(level);
39      }
40  
41      @Override
42      public Action getPlayerAction(Level level) {
43          draw(level);
44          Action playerAction = input.getAction();
45          if (checkForGameCommands(playerAction)) {
46              return Action.NONE;
47          }
48          return playerAction;
49      }
50  
51      private boolean checkForGameCommands(Action playerAction) {
52          if (playerAction.equals(Action.QUIT)) {
53              //TODO: nicer quit
54              System.exit(0);
55          }
56          return false;
57      }
58  
59      @Override
60      public void showMessage(String message) {
61          visualInterface.showMessage(message);
62      }
63  
64  }
