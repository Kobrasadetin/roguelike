6   package com.majesticbit.roguelike.ascii;
7   
8   import com.majesticbit.roguelike.domain.Dungeon;
9   
10  /**
11   *
12   * @author Master
13   */
14  public class AsciiViewer {
15  
16      private int width, height;
17      private String baseString;
18      private Dungeon dungeon;
19  
20      public AsciiViewer(Dungeon dungeon) {
21          this.width = dungeon.getWidth();
22          this.height = dungeon.getHeight();
23          this.dungeon = dungeon;
24          initialize();
25      }
26  
27      public String draw() {
28          StringBuilder output = new StringBuilder();
29          StringBuilder line = new StringBuilder(baseString);
30          for (int y = 0; y < height; y++) {
31              for (int x = 0; x < width; x++) {
32                  if (dungeon.getTile(x, y).isSolid()) {
33                      line.setCharAt(x, ' ');
34                  }
35                  if (dungeon.getTile(x, y).isWall()) {
36                      line.setCharAt(x, '#');
37                  }
38                  if (!dungeon.getTile(x, y).isSolid()) {
39                      line.setCharAt(x, '.');
40                  }
41              }
42              output.append(line);
43              output.append('\n');
44          }
45          return output.toString();
46      }
47  
48      private void initialize() {
49          StringBuilder dottedRow = new StringBuilder();
50          for (int x = 0; x < width; x += 8) {
51              dottedRow.append("........");
52          }
53          dottedRow.setLength(width);
54          baseString = dottedRow.toString();
55      }
56  
57  }
