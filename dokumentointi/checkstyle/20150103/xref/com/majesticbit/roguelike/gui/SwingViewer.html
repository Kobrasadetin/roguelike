6   package com.majesticbit.roguelike.gui;
7   
8   import com.majesticbit.roguelike.domain.Position;
9   import com.majesticbit.roguelike.domain.dungeon.Dungeon;
10  import com.majesticbit.roguelike.domain.dungeon.Tile;
11  import com.majesticbit.roguelike.domain.simulation.DynamicObject;
12  import com.majesticbit.roguelike.domain.level.Level;
13  import java.awt.Font;
14  import squidpony.squidgrid.gui.SGPane;
15  import squidpony.squidgrid.gui.swing.SwingPane;
16  
17  /**
18   *
19   * @author Master
20   */
21  public class SwingViewer extends VisualInterface {
22  
23      private int width, height;
24      private SGPane pane;
25  
26      public SwingViewer() {
27          pane = new SwingPane();
28          pane.initialize(30, 20, new Font("TimesRoman", Font.PLAIN, 18));
29      }
30  
31      @Override
32      public void draw(Level level) {
33          pane.placeText(0, 0, characterMap(level));
34  
35      }
36  
37      public static char[][] characterMap(Level level) {
38  
39          Dungeon dungeon = level.getDungeon();
40          int width = dungeon.getWidth();
41          int height = dungeon.getHeight();
42          char[][] map = new char[width][height];
43  
44          for (int y = 0; y < height; y++) {
45              for (int x = 0; x < width; x++) {
46                  Tile tile = dungeon.getTile(x, y);
47                  map[x][y] = tile.getDescription().getSymbol();
48              }
49          }
50  
51          // draw creatures and other dynamic objects
52          for (DynamicObject object : level.getObjects()) {
53              Position pos = object.getPosition();
54              map[pos.x][pos.y] = object.getDescription().getSymbol();
55          }
56  
57          return map;
58      }
59  
60  }
