6   package com.majesticbit.roguelike.gui;
7   
8   import com.majesticbit.roguelike.domain.Game;
9   import com.majesticbit.roguelike.domain.creatures.actions.Action;
10  import com.majesticbit.roguelike.domain.level.Level;
11  
12  /**
13   *
14   * @author Master
15   */
16  public class UserInterface {
17  
18      private VisualInterface visualInterface;
19      private PlayerInput input;
20      private boolean quitGame = false;
21  
22      public UserInterface() {
23          this.visualInterface = new AsciiViewer();
24          this.input = new SystemInput();
25      }
26  
27      public void draw(Level level) {
28          visualInterface.draw(level);
29      }
30  
31      public Action getPlayerAction() {
32          Action playerAction = input.getAction();
33          if (checkForGameCommands(playerAction))
34          {
35              return Action.NONE;
36          }
37          return playerAction;
38      }
39  
40      public boolean playerWantsToQuit() {
41          return quitGame;
42      }
43  
44      private boolean checkForGameCommands(Action playerAction) {
45          if (playerAction.equals(Action.QUIT)) {
46              quitGame = true;
47              return true;
48          }
49          return false;
50      }
51  
52  }
