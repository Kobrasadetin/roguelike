18   *
19   * @author Master
20   */
21  public class Game {
22  
23      private BasicLevel currentLevel;
24      private Player player;
25      private VisibilitySolver visibilitySolver;
26  
27      private UserInterface ui;
28  
29      public Game(UserInterface ui) {
30          this.ui = ui;
31          this.player = new Player(ui);
32      }
33  
34      public Level getCurrentLevel() {
35          return currentLevel;
36      }
37  
38      public Player getPlayer() {
39          return player;
40      }
41  
42      public void play() {
43          currentLevel = createNewLevel();
44          createPlayterAvatar();
45          while (!ui.playerWantsToQuit()) {
46              currentLevel.advanceOneTimestep();
47          }
48          //player.bestowFullKnowledge(currentLevel);
49      }
50  
51      private BasicLevel createNewLevel() {
52          Dungeon dungeon = new DungeonBuilder().toDungeon();
53          BasicLevel newLevel = new BasicLevel(dungeon);
54          return newLevel;
55      }
56  
57      private void advanceOneTimestep() {
58          currentLevel.advanceOneTimestep();
59      }
60  
61      private void createPlayterAvatar() {
62          Humanoid playerCreature = new Humanoid(new Position(4, 4), new TextDescription("player", '@'));
63          playerCreature.setController(player);
64          currentLevel.addCreature(playerCreature, playerCreature.getPosition());
65      }
66  
67  }
